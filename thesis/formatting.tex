% Chapter styling without line, for toc, bib, etc.

{\chapter}[block]{\sc}{\centering Chapter \ \thechapter}{0.1in} 
{
    \centering 
    \vspace{0.1in}
    \centering\bfseries\large
} 
[
\vspace{-1ex}
%\rule{\textwidth}{0.3pt}
]
}

\newcommand{\chapstylewithline}{

    \centering\bfseries\large
} 
[

%\rule{\textwidth}{0.3pt}
]
}
