% Chapter styling without line for frontmatter and backmatter.  
\newcommand{\chapstylenoline}{\titleformat
{\chapter}[block]{}{}{0.1in} 
{
    \centering 
    \vspace{0.1in}
    \centering\bfseries\large
} 
[
\vspace{-1ex}
%\rule{\textwidth}{0.3pt}
]
}

% Chapter styling with line, for midmatter chapters. 
\newcommand{\chapstylewithline}{
	\titleformat
	{\chapter}[block]{}{\centering {\sc Chapter} \ \thechapter}{0.1in} 
	{
		\centering 
		\\\rule{12cm}{0.9pt}\\
		\vspace{0.1in}
		\centering\bfseries\large
	} 
	[
	\vspace{-1ex}
	%\rule{\textwidth}{0.3pt}
	]
}



%% Export Various Stylings for frontmatter, midmatter and backmatter


% Export front matter styling
\newcommand{\frontmatterstyling}{
\chapstylenoline
\pagenumbering{roman}
\pagestyle{fancy}
\renewcommand{\headrulewidth}{0pt}
\cfoot{\thepage}
}

% Mid matter styling -- Chapter and Section
\newcommand{\headingstyle}{%
\pagestyle{fancy}%
\renewcommand{\chaptermark}[1]{ \markboth{Chap. \thechapter. \;##1}{} }%
\renewcommand{\sectionmark}[1]{ \markright{{\em ##1}}{} }%
\lhead{\leftmark}%
\rhead{\rightmark}%
\renewcommand{\headrulewidth}{0.5pt}
}

\newcommand{\midmatterstyling}{%
\onehalfspacing
%\newgeometry{twoside,margin=1.2in}
%\doublespacing
\headingstyle
\pagenumbering{arabic}
\let\secsymbol\S
\titleformat{\section}[block]{\normalfont\bfseries}{\secsymbol\thesection.}{0.5em}{\centering}[]
\titleformat{\subsection}[block]{\normalfont\it\bfseries}{\thesubsection.}{0.5em}{\centering}%[\vspace{-0.7cm}\rule{\titlewidth}{0.5pt}\vspace{0.3cm}]
\chapstylewithline
}

\newcommand{\chapterquote}[2]{
\begin{adjustwidth}{12em}{2em}
\begin{flushright}
\textit{#1} \\
\vspace{0.1cm}
--- #2
\end{flushright}
\end{adjustwidth}
}


% Export backmatter styling
\newcommand{\backmatterstyling}{%
\pagestyle{plain}
\singlespacing%
\chapstylenoline%
\renewcommand{\chaptermark}[1]{ \markboth{##1}{} }%
}%
